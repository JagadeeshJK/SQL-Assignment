1.SELECT DEPT.DNAME, EMP.ENAME
FROM EMP
RIGHT OUTER JOIN DEPT
ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.ENAME IS NULL

2.SELECT REPORTEE.ENAME, JOB
FROM EMP REPORTEE
WHERE REPORTEE.MGR IS NULL

3.SELECT E.ENAME, E.JOB, E.COMM, D.LOC
FROM EMP E, DEPT D
WHERE D.LOC = 'NEW YORK' AND E.COMM>1000

4.SELECT COUNT(D.LOC) AS 'NO OF PEOPLE'
FROM EMP E, DEPT D
WHERE D.LOC LIKE 'CHICAGO' AND E.DEPTNO = D.DEPTNO

5.SELECT E.ENAME, D.LOC
FROM EMP E, DEPT D
WHERE D.LOC LIKE 'CHICAGO' AND E.DEPTNO = D.DEPTNO

6.SELECT E.ENAME, D.LOC
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO
ORDER BY D.LOC

7.SELECT COUNT(E.EMPNO) AS 'NO OF EMPLOYEES', COUNT(DISTINCT(D.DEPTNO)) AS 'NO OF DEPARTMENTS'
FROM EMP E RIGHT JOIN DEPT D
ON E.DEPTNO = D.DEPTNO

8.SELECT E.ENAME, F.ENAME 'MANAGER', F.HIREDATE 'MANAGER HIREDATE', E.HIREDATE
FROM EMP E, EMP F
WHERE E.MGR = F.EMPNO AND E.HIREDATE<F.HIREDATE

9.SELECT COUNT(MGR) AS 'NO OF MANAGERS'
FROM EMP
WHERE JOB = 'MANAGER'

10.SELECT REPORTEE.ENAME, JOB
FROM EMP REPORTEE
WHERE REPORTEE.MGR IS NULL

11.SELECT E.ENAME, D.DEPTNO
FROM EMP E, DEPT D
WHERE E.DEPTNO = D.DEPTNO AND E.ENAME = 'SMITH'

12.SELECT E.ENAME, D.LOC, E.JOB
FROM EMP E, DEPT D
WHERE E.DEPTNO=D.DEPTNO AND D.LOC <> 'BOSTON' AND E.JOB = 'SALESMAN'

13.SELECT E.ENAME, E.EMPNO, E.JOB, E.MGR, E.HIREDATE, E.COMM, D.DEPTNO, D.LOC
FROM EMP E JOIN DEPT D
ON E.DEPTNO = D.DEPTNO

14.SELECT ENAME, SAL
FROM EMP
WHERE JOB = 'SALESMAN' AND SAL<(SELECT SAL
                                FROM EMP
				WHERE ENAME = 'MILLER')

15.SELECT *
FROM EMP E, DEPT D
WHERE D.LOC LIKE 'CHICAGO' AND E.DEPTNO = D.DEPTNO

16.SELECT ENAME
FROM EMP
WHERE SAL>(SELECT MIN(SAL)
FROM EMP
WHERE JOB = 'MANAGER')

17.SELECT ENAME, JOB
FROM EMP
WHERE DEPTNO=(SELECT DEPTNO
FROM DEPT
WHERE DNAME = 'ACCOUNTING')

18.SELECT ENAME 
FROM EMP
WHERE JOB = 'MANAGER'

19.SELECT DNAME,
(CASE 
WHEN DNAME != 'SALES'
THEN 'NOTSALES' ELSE 'SALES' END)
FROM DEPT

20.SELECT SAL,
CASE
WHEN SAL > 1000 THEN SAL
ELSE 1000
END AS 'DEFAULT SALARY'
FROM EMP

21.SELECT SAL,
CASE
WHEN SAL < 1000 THEN 'LESS'
WHEN SAL > 2500 THEN 'HIGH'
    ELSE 'MEDIUM'
END AS 'DESC'
FROM EMP
